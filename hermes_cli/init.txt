CREATE TABLE "account" (
	"acc"	text NOT NULL,
	"name"	text NOT NULL,
	"phone"	text NOT NULL,
	"package"	int NOT NULL,
	"username"	text NOT NULL,
	"password"	text NOT NULL,
	"install_date"	text,
	"balance"	decimal(10, 2) DEFAULT 0.00,
	PRIMARY KEY("acc")
);

CREATE TABLE "contacts" (
	"cid"	INTEGER NOT NULL,
	"account"	TEXT NOT NULL,
	"contact"	TEXT NOT NULL,
	PRIMARY KEY("cid")
);

CREATE TABLE "finances" (
	"fid"	int NOT NULL,
	"acc"	text NOT NULL,
	"money in"	int,
	"money out"	int,
	"date"	TEXT,
	"description"	text,
	PRIMARY KEY("fid")
);

CREATE TABLE "package" (
	"pno"	int NOT NULL,
	"name"	text,
	"speed"	int NOT NULL,
	"days"	INTEGER NOT NULL,
	"max_users"	int,
	"price"	decimal(8, 2) NOT NULL DEFAULT 00.00,
	"type"	TEXT NOT NULL,
	PRIMARY KEY("pno")
);

CREATE TABLE "payments" (
	"pid"	INTEGER NOT NULL UNIQUE,
	"acc"	text NOT NULL,
	"code"	text NOT NULL UNIQUE,
	"amount"	int,
	"source"	text,
	"date"	text,
	"time"	text,
	PRIMARY KEY("pid" AUTOINCREMENT)
);

CREATE TABLE "sessions" (
	"sid"	int NOT NULL,
	"acc"	text NOT NULL,
	"profile"	text,
	"start date"	text,
	"start time"	text,
	"end date"	text,
	"end time"	text,
	"status"	text NOT NULL,
	"creation date"	TEXT NOT NULL,
	PRIMARY KEY("sid")
);

CREATE TABLE "PppoeAccounts" (
	"acc"	TEXT NOT NULL UNIQUE,
	"name"	TEXT NOT NULL,
	"phone"	TEXT NOT NULL,
	"location"	TEXT NOT NULL,
	"ip"	TEXT,
	"username"	TEXT NOT NULL,
	"password"	TEXT NOT NULL,
	"install date"	TEXT,
	"package"	INTEGER NOT NULL,
	"balance"	INTEGER NOT NULL,
	PRIMARY KEY("acc")
);
